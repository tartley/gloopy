        glGenVertexArrays(1, vao_id)

        try:
            GL.glEnableClientState(GL.GL_VERTEX_ARRAY)
            GL.glVertexPointer(

            )
            GL.glColorPointer(
                Color.COMPONENTS, GL.GL_FLOAT, self.stride, self.vbo + 12
            )
            GL.glNormalPointer(
                GL.GL_FLOAT, self.stride, self.vbo + 24
            )
        finally:
            glBindVertexArray(0)
